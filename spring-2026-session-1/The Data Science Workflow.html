Steps in a Typical Data Science Project

Image courtesy of R4DS (2e), Wickham et al

Other forms: PPDAC

The data science workflow, as described in R for Data Science (2e) by the aforementioned Hadley Wickham, consists of how information is transformed from raw data into actionable insights. A full data solutions workflow consists of the following steps:

  1. Import: Identify sources of and methods to acquire data, utilizing automation wherever possible.
  2. Tidy: Wrangle data into a tidy format conducive for understanding its properties and store it in an accessible and secure location. Should be performed autonomously wherever possible.
  3. Transform: Extract features of interest from data and perform additional computations to augment the dataset.
  4. Visualize: Generate informative illustrations of the transformed data in order to provide additional context to insights.
  5. Model: Apply statistical methods, machine learning, and artificial intelligence to understand data via prediction, identifying trends, and detecting anomalies.
  6. Communicate: Interpret results and produce reports, web applications, and other compelling summaries of finds in a way that is digestible and actionable for your stakeholders.

# tidy the Elo data

def normalize_team_name(series):
    return (
        series.astype(str)
        .str.replace('\xa0', ' ', regex=False)
        .str.replace(r'\s+', ' ', regex=True)
        .str.strip()
    )

most_recent_elo = (
    elo_data
    .assign(
        date = pd.to_datetime(elo_data['date'], format="%m/%d/%Y", errors='coerce'),
        team_key = lambda df: normalize_team_name(df['team'])
    )
    .sort_values('date')
    .drop_duplicates(subset='team_key', keep='last')
    .loc[:, ['team_key', 'rating']]
)

fifa_teams = (
    fifa_teams
    .assign(team_name = lambda df: df['team_name'].str.strip())
    .replace({
        "Winner UEFA Playoff A": "Bosnia and Herzegovina",
        "Winner UEFA Playoff B": "Sweden",
        "Winner UEFA Playoff C": "Turkey",
        "Winner UEFA Playoff D": "Czechia",

        "USA": "United States",

        "Cabo Verde": "Cape Verde",
        "Côte d'Ivoire": "Ivory Coast"
    })
    .assign(team_key = lambda df: normalize_team_name(df['team_name']))
)

# simulate the group stage

simulate_group_stage <- function(schedule_data) {
  schedule_data %>% 
    mutate(
      home_win_prob = elo_prob(rating_home, rating_away),
      home_team_wins = rbinom(n(), size = 1, prob = home_win_prob),
      home_team_points = case_when(
        home_team_wins == 1 ~ 3,
        home_team_wins == 0 ~ 0
      ),
      away_team_points = case_when(
        home_team_wins == 1 ~ 0,
        home_team_wins == 0 ~ 3
      )
    )
}

set.seed(2026)
n_simulations = 1000

simulations <- 
  map_dfr(
    1:n_simulations, 
  ~ simulate_group_stage(complete_schedule) %>% 
    mutate(simulation_id = .x)
)

simulated_standings <- 
  simulations %>% 
  select(simulation_id, group = group_letter_home, team = team_name_home, pts = home_team_points) %>% 
  group_by(simulation_id, group, team) %>% 
  summarise(total_points = sum(pts)) %>% 
  group_by(simulation_id, group) %>% 
  arrange(desc(total_points)) %>% 
  mutate(rank = row_number()) %>%
  ungroup()

# the top 8 teams in each group will advance to the knockout stage
top_two <- 
  simulated_standings %>% 
  filter(rank <= 2)

# the top 8 third-place teams will aslo advance
best_third <- 
  simulated_standings %>% 
  filter(rank == 3) %>% 
  group_by(simulation_id) %>% 
  arrange(desc(total_points)) %>% 
  slice_head(n = 8) %>% 
  ungroup()

advancing_teams <- bind_rows(top_two, best_third) %>% 
  count(team, name = "times_advanced") %>% 
  mutate(prob_advance = times_advanced / n_simulations) %>% 
  arrange(desc(prob_advance))

Python
import numpy as np

def simulate_group_stage(schedule_data):
    schedule_data = schedule_data.copy()
    schedule_data['home_win_prob'] = elo_prob(schedule_data['rating_home'], schedule_data['rating_away'])
    schedule_data['home_team_wins'] = np.random.binomial(1, schedule_data['home_win_prob'])
    schedule_data['home_team_points'] = np.where(schedule_data['home_team_wins'] == 1, 3, 0)
    schedule_data['away_team_points'] = np.where(schedule_data['home_team_wins'] == 1, 0, 3)
    return schedule_data
  
# set seed
np.random.seed(2026)

n_simulations = 10000

simulations = pd.concat(
    [simulate_group_stage(complete_schedule).assign(simulation_id=i) for i in range(1, n_simulations + 1)],
    ignore_index=True
)

simulated_standings = (
    simulations
    .loc[:, ['simulation_id', 'group_letter_home', 'team_name_home', 'home_team_points']]
    .rename(columns={'group_letter_home': 'group', 'team_name_home': 'team', 'home_team_points': 'pts'})
    .groupby(['simulation_id', 'group', 'team'])
    .agg(total_points=('pts', 'sum'))
    .reset_index()
    .sort_values(['simulation_id', 'group', 'total_points'], ascending=[True, True, False])
    .assign(rank=lambda df: df.groupby(['simulation_id', 'group']).cumcount() + 1)
)

# the top 2 teams in each group will advance to the knockout stage
top_two = (
    simulated_standings
    .query('rank <= 2')
    .reset_index(drop=True)
)

# the top 8 third-place teams will also advance
best_third = (
    simulated_standings
    .query('rank == 3')
    .groupby('simulation_id')
    .apply(lambda df: df.nlargest(8, 'total_points'))
    .reset_index(drop=True)
)

advancing_teams = (
    pd.concat([top_two, best_third], ignore_index=True)
    .groupby('team')
    .agg(times_advanced=('team', 'size'))
    .assign(prob_advance=lambda df: df['times_advanced'] / n_simulations)
    .sort_values('prob_advance', ascending=False)
    .reset_index()
)

print(advancing_teams.head())
