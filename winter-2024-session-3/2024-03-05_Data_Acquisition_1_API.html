Simply pass the proper data frame to woba_plus:

  dplyr::filter(PA > 400) %>%
  woba_plus %>%
  dplyr::arrange(desc(wOBA)) %>%
  dplyr::select(Name, Team, season, PA, wOBA, wOBA_CON) %>%
